### 我的解法

```CS{.line-numbers}
public class Solution {
    public int LengthOfLongestSubstring(string s) {
        List<Char> charList = new List<Char>();
        int minWidth=0;
        //遍历字符数组
        foreach(char item in s.ToCharArray())
        {              
            //判断链表元素是否重复
            if(charList.Count!=0 && charList.Contains(item))  
            {
                minWidth=charList.Count>minWidth?charList.Count:minWidth;
                int index=charList.IndexOf(item);
                //在链表中移除从0开始到重复字符的第一个下标所有的元素
                charList.RemoveRange(0,index+1);
            }
            charList.Add(item);
        }
        return charList.Count>minWidth?charList.Count:minWidth;
    }
}
```
